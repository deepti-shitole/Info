pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
               checkout scmGit(branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[credentialsId: 'github-cred1', url: 'https://github.com/deepti-shitole/Python_list.git']])
            }
        }
        stage('Build') {
            steps {
            git branch: 'main', credentialsId: 'github-cred1', url: 'https://github.com/deepti-shitole/Python_list.git'
            bat 'python list.py'
            }
        }
        stage ('test') {
            steps {
                echo "the job succeed"
            }
        }
        
    }
}
